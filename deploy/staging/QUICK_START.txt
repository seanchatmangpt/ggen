━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

1. START THE CLUSTER (single command)

   $ cd /Users/sac/ggen
   $ ./deploy/staging/start.sh

   Expected: "✓ STAGING DEPLOYMENT SUCCESSFUL" after ~60-90s

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

2. VERIFY CLUSTER RUNNING

   $ docker-compose -f deploy/staging/docker-compose.yml ps

   Expected: All 5 containers showing "healthy" or "running"

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

3. ACCESS DASHBOARDS

   Prometheus:         http://localhost:9090
   ggen Node 1:        http://localhost:8080/health
   ggen Node 2:        http://localhost:8081/health
   ggen Node 3:        http://localhost:8082/health
   Firestore Emulator: localhost:8081

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

4. KILL A NODE (CHAOS TESTING)

   $ docker-compose -f deploy/staging/docker-compose.yml kill ggen-node-1

   Watch other nodes handle failure. Resurrect with:

   $ docker-compose -f deploy/staging/docker-compose.yml up -d ggen-node-1

━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

5. STOP THE CLUSTER

   $ docker-compose -f deploy/staging/docker-compose.yml down

   Or nuke everything (delete volumes):

   $ docker-compose -f deploy/staging/docker-compose.yml down -v

TROUBLESHOOTING

  Containers won't start?
    → docker logs ggen-node-1
    → Check ports: lsof -i :8080

  Health checks failing?
    → docker-compose down -v && ./deploy/staging/start.sh

  Prometheus not scraping?
    → docker logs prometheus
