USAGE EXAMPLE: SMART METER WITH MULTIPLE ONTOLOGIES
================================================================================

@prefix saref: <https://saref.etsi.org/saref#> .
@prefix schema: <https://schema.org/> .
@prefix dcat: <https://www.w3.org/ns/dcat#> .
@prefix dcterms: <https://purl.org/dc/terms/> .
@prefix prov: <https://www.w3.org/ns/prov#> .

:Meter_Building_A a saref:Device ;
    rdfs:label "Smart Meter - Building A" ;
    schema:manufacturer "Siemens AG" ;
    saref:consumes :Electricity ;
    dcterms:issued "2024-01-01"^^xsd:date .

:MeteringData_Daily a dcat:Dataset ;
    dcterms:title "Daily Metering Data for Building A" ;
    dcat:keyword "consumption", "smart meter", "real-time" ;
    prov:wasGeneratedBy :Meter_Building_A ;
    prov:wasAttributedTo :UtilityCompany_X ;
    dcat:distribution [
        dcat:accessURL <https://api.utility.com/data/meter_A> ;
        dcat:mediaType "application/json" ;
    ] .
