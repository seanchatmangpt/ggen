\chapter{Implementation Artifacts}

This appendix describes configuration files and schemas.

\section{ggen.toml Schema}

\begin{minted}[fontsize=\small]{toml}
# ggen configuration file
[general]
spec_dir = ".specify/specs"
output_dir = "generated"
audit_log = "audit.log"

[pipeline]
stages = ["normalize", "extract", "emit", "canonicalize", "receipt"]

[normalization]
format = "turtle"
validation = "shacl"
entropy_threshold = 20

[extraction]
query_dir = ".specify/queries"
patterns = ["entity", "relationship", "constraint"]

[emission]
template_dir = ".specify/templates"
output_encoding = "utf-8"

[canonicalization]
rust_cmd = "rustfmt"
typescript_cmd = "prettier"
python_cmd = "black"

[receipt]
key_file = ".ggen/private-key.pem"
hash_algorithm = "blake3"
signature_algorithm = "ed25519"

[testing]
coverage_target = 80
mutation_target = 60
determinism_iterations = 10000
\end{minted}

\section{Template Syntax}

Tera template syntax:

\begin{minted}[fontsize=\small]{jinja2}
# Comments: {# comment #}

# Variables
{{ entity.name }}
{{ field.type }}

# Loops
{% for field in entity.fields %}
pub {{ field.name }}: {{ field.rust_type }},
{% endfor %}

# Conditionals
{% if field.optional %}
Option<{{ field.rust_type }}>,
{% else %}
{{ field.rust_type }},
{% endif %}

# Filters
{{ entity.name | lower }}
{{ field.type | snake_case }}

# Macros
{% macro rust_type(field) -%}
  {%- match field.type -%}
    {%- when "string" -%}
      String
    {%- when "integer" -%}
      i64
    {%- when "boolean" -%}
      bool
    {%- else -%}
      String
  {%- endmatch -%}
{%- endmacro %}

{{ self::rust_type(field) }}
\end{minted}
