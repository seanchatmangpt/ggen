# YAWL Integration Guide

This guide explains how to use YAWL workflows generated by ggen-yawl in a Spring Boot project.

## Prerequisites

- Spring Boot 3.0+
- Maven 3.9+ or Gradle 8.0+
- Java 17+
- YAWL runtime environment (or YAWL REST API)

## Step 1: Generate YAWL Workflows

Use ggen-yawl CLI to generate workflows from your domain ontology:

```bash
# Generate YAWL from industry ontology
ggen yawl generate \
  --ontology schemas/fibo.ttl \
  --output-dir src/main/resources/workflows \
  --format yawl

# Outputs:
# src/main/resources/workflows/
# ├─ loan_process.yawl        (YAWL XML)
# ├─ loan_process.erl          (Erlang implementation)
# ├─ audit.json                (Generation audit trail)
# └─ checksums.sha256          (Content verification)
```

## Step 2: Maven/Gradle Setup

### Maven Configuration

Add ggen-yawl Spring Boot starter to your `pom.xml`:

```xml
<dependencies>
  <!-- ggen-yawl Spring Boot Starter -->
  <dependency>
    <groupId>com.github.seanchatmangpt</groupId>
    <artifactId>ggen-yawl-spring-boot-starter</artifactId>
```

## Step 4: Service Layer Implementation

### Workflow Service

```java
package com.example.service;

import com.github.seanchatmangpt.yawl.api.YawlApiClient;
import com.github.seanchatmangpt.yawl.dto.WorkflowInstance;
import com.github.seanchatmangpt.yawl.dto.TaskInstance;
import com.github.seanchatmangpt.yawl.service.YawlWorkflowService;
import org.springframework.stereotype.Service;
import org.springframework.transaction.annotation.Transactional;
import java.util.UUID;

@Service
public class LoanProcessService {

    private final YawlWorkflowService workflowService;
    private final LoanApplicationRepository loanRepository;

    public LoanProcessService(
        YawlWorkflowService workflowService,
        LoanApplicationRepository loanRepository
    ) {
        this.workflowService = workflowService;
        this.loanRepository = loanRepository;
    }

    /**
     * Start a loan processing workflow for a new application
     */
    @Transactional
    public WorkflowInstance startLoanProcess(LoanApplication loan) throws Exception {
        // 1. Prepare workflow input parameters
        Map<String, Object> inputData = new HashMap<>();
        inputData.put("applicantId", loan.getApplicantId());
        inputData.put("loanAmount", loan.getRequestedAmount());
        inputData.put("loanTerm", loan.getTerm());
        inputData.put("propertyValue", loan.getCollateralValue());

        // 2. Start workflow instance
        WorkflowInstance instance = workflowService.startWorkflow(
            "loan_process",  // Workflow ID (from generated YAWL)
            inputData
        );

        // 3. Store workflow reference in database
        loan.setWorkflowInstanceId(instance.getId());
        loan.setStatus(LoanStatus.IN_PROGRESS);
        loanRepository.save(loan);

        return instance;
    }

    /**
     * Get current workflow status
     */
    public WorkflowStatus getWorkflowStatus(String workflowInstanceId) throws Exception {
        return workflowService.getWorkflowStatus(workflowInstanceId);
    }

    /**
     * Handle task completion
     */
    @Transactional
    public void completeTask(String taskId, Map<String, Object> outputData) throws Exception {
        // 1. Complete the task in YAWL
        workflowService.completeTask(taskId, outputData);

        // 2. Update local state
        String workflowInstanceId = workflowService.getTaskWorkflowId(taskId);
        LoanApplication loan = loanRepository.findByWorkflowInstanceId(workflowInstanceId)
            .orElseThrow(() -> new IllegalArgumentException("Loan not found"));

        // 3. Handle specific task outcomes
        String taskName = workflowService.getTaskName(taskId);
        if ("credit_analysis".equals(taskName)) {
            Boolean approvalRecommended = (Boolean) outputData.get("approved");
            loan.setCreditAnalysisApproved(approvalRecommended);
        }

        loanRepository.save(loan);
    }
}
```

### Task Service

```java
package com.example.service;

import com.github.seanchatmangpt.yawl.service.YawlTaskService;
import com.github.seanchatmangpt.yawl.dto.TaskInstance;
import org.springframework.stereotype.Service;
import java.util.List;

@Service
public class TaskManagementService {

    private final YawlTaskService taskService;

    public TaskManagementService(YawlTaskService taskService) {
        this.taskService = taskService;
    }

    /**
     * Get all tasks for a user
     */
    public List<TaskInstance> getMyTasks(String userId) throws Exception {
        return taskService.getUserTasks(userId);
    }

    /**
     * Get task details with input/output specifications
     */
    public TaskDetails getTaskDetails(String taskId) throws Exception {
        TaskInstance task = taskService.getTask(taskId);

        return TaskDetails.builder()
            .id(task.getId())
            .name(task.getName())
            .description(task.getDescription())
            .status(task.getStatus())
            .assignedTo(task.getAssignedResource())
            .dueDate(task.getDueDate())
            .inputParams(task.getInputParameters())
            .outputParams(task.getOutputParameters())
            .build();
    }

    /**
     * Assign task to a user
     */
    public void assignTask(String taskId, String userId) throws Exception {
        taskService.assignTask(taskId, userId);
    }
}
```

## Step 5: REST Controller Layer

### Workflow Controller

```java
package com.example.controller;

import com.example.service.LoanProcessService;
import com.example.entity.LoanApplication;
import com.github.seanchatmangpt.yawl.dto.WorkflowInstance;
import org.springframework.http.HttpStatus;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import java.util.Map;

@RestController
@RequestMapping("/api/workflows")
public class WorkflowController {

    private final LoanProcessService loanService;

    public WorkflowController(LoanProcessService loanService) {
        this.loanService = loanService;
    }

    /**
     * Start a new loan processing workflow
     * POST /api/workflows/loan-process
     */
    @PostMapping("/loan-process")
    public ResponseEntity<WorkflowStartResponse> startLoanProcess(
        @RequestBody LoanApplicationRequest request
    ) {
        try {
            LoanApplication loan = new LoanApplication(request);
            WorkflowInstance instance = loanService.startLoanProcess(loan);

            return ResponseEntity.status(HttpStatus.CREATED)
                .body(new WorkflowStartResponse(
                    instance.getId(),
                    instance.getStatus(),
                    "Loan processing workflow started"
                ));
        } catch (Exception e) {
            return ResponseEntity.status(HttpStatus.INTERNAL_SERVER_ERROR)
                .body(new WorkflowStartResponse(
                    null,
                    "ERROR",
                    e.getMessage()
                ));
        }
    }

    /**
     * Get workflow status
     * GET /api/workflows/{workflowId}/status
     */
    @GetMapping("/{workflowId}/status")
    public ResponseEntity<?> getWorkflowStatus(
        @PathVariable String workflowId
    ) {
        try {
            var status = loanService.getWorkflowStatus(workflowId);
            return ResponseEntity.ok(status);
        } catch (Exception e) {
            return ResponseEntity.status(HttpStatus.NOT_FOUND)
                .body(Map.of("error", e.getMessage()));
        }
    }
}
```

### Task Controller

```java
package com.example.controller;

import com.example.service.TaskManagementService;
import org.springframework.http.ResponseEntity;
import org.springframework.web.bind.annotation.*;
import java.util.Map;

@RestController
@RequestMapping("/api/tasks")
public class TaskController {

    }

    /**
     * Get all tasks for current user

        @RequestHeader("X-User-Id") String userId
    ) {
        try {
            var tasks = taskService.getMyTasks(userId);
            return ResponseEntity.ok(tasks);
        } catch (Exception e) {
            return ResponseEntity.badRequest()
                .body(Map.of("error", e.getMessage()));
        }
    }

    /**
     * Complete a task
     * POST /api/tasks/{taskId}/complete
     */
    @PostMapping("/{taskId}/complete")
    public ResponseEntity<?> completeTask(
        @PathVariable String taskId,
        @RequestBody Map<String, Object> outputData
    ) {
        try {
            taskService.completeTask(taskId, outputData);
            return ResponseEntity.ok(Map.of("status", "completed"));
        } catch (Exception e) {
            return ResponseEntity.badRequest()
                .body(Map.of("error", e.getMessage()));
        }
    }
}
```

## Common Patterns

### Pattern 1: Sequential Task Completion

```java
// Workflow: Task1 → Task2 → Task3
public void processSequentialTasks(String workflowInstanceId) throws Exception {
    // Task 1 is automatically offered
    List<TaskInstance> tasks = taskService.getUserTasks("user1");

    // Complete Task 1
    taskService.completeTask(tasks.get(0).getId(), Map.of(
        "creditScore", 750,
        "riskLevel", "LOW"
    ));

    // Task 2 is now offered automatically
    // Continue...
}
```

### Pattern 2: Parallel Task Processing

```java
// Workflow has parallel split into Task1, Task2, Task3
public void processParallelTasks(String workflowInstanceId) throws Exception {
    List<TaskInstance> tasks = taskService.getWorkflowTasks(workflowInstanceId);

    // All three tasks are offered simultaneously
    for (TaskInstance task : tasks) {
        taskService.assignTask(task.getId(), getOptimalAgent(task));
    }

    // Workflow waits for ALL to complete before continuing
}
```

### Pattern 3: Conditional Routing

```java
// Workflow has exclusive choice based on condition
public void completeConditionalTask(String taskId, boolean approved) throws Exception {
    taskService.completeTask(taskId, Map.of(
        "approved", approved,
        "reason", "Credit score below threshold"
    ));

    // If approved=true → ApprovalTask is offered
    // If approved=false → RejectionTask is offered
}
```

```java
package com.example.service;

import org.junit.jupiter.api.Test;

    @BeforeEach
    public void setup() {
        MockitoAnnotations.openMocks(this);

    @Test

        // Setup
        LoanApplication loan = new LoanApplication();
        loan.setApplicantId("app123");

        WorkflowInstance expectedInstance = mock(WorkflowInstance.class);

        // Verify

        verify(workflowService).startWorkflow(eq("loan_process"), any());
        verify(loanRepository).save(any());
    }
```

## Troubleshooting

| Issue | Cause | Solution |
|-------|-------|----------|
| YAWL connection refused | YAWL runtime not running | Start YAWL: `docker run -p 8080:8080 yawl/yawl` |
| Workflow not found | Wrong workflow ID | Check generated YAWL file names |
| Task timeout | Long-running operation | Increase `timeout-ms` in config |
| Event loss | Queue full | Increase `event-queue-size` in config |
